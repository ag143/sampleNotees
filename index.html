Kubernetes

  • 🚀 **Pods, Deployments, Services**
  • 🎭 **Helm & Operators**

apiVersion: apps/v1
kind: Deployment
metadata:
  name: my-app
spec:
  replicas: 3
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
        - name: my-container
          image: nginx

  • ✅ **GitHub Actions, Jenkins, GitLab CI**
  • 🔄 **ArgoCD, Spinnaker**

name: CI Pipeline
on: [push]
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v3
      - name: Run Tests
        run: echo "Running tests..."
